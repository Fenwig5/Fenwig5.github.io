function checkCashRegister(price, cash, cashInDrawer) { var startChange = cash - price; var change = cash - price; var cid = cashInDrawer; var exactChange = []; var ecAvail; //label sections of cid array // Example cash-in-drawer array: var pennies = cid[0][1]; // [["PENNY", 1.01], var nickels = cid[1][1]; // ["NICKEL", 2.05], var dimes = cid[2][1]; // ["DIME", 3.1], var quarters = cid[3][1]; // ["QUARTER", 4.25], var ones = cid[4][1]; // ["ONE", 90], var fives = cid[5][1]; // ["FIVE", 55], var tens = cid[6][1]; // ["TEN", 20], var twenties = cid[7][1]; // ["TWENTY", 60], var hundreds = cid[8][1]; // ["ONE HUNDRED", 100]] var cidTotal = pennies+nickels+dimes+quarters+ones+fives+tens+twenties+hundreds; //make counters to push the right amount in one subarray to exactChange var penniesCount = 0; var nickelsCount = 0; var dimesCount = 0; var quartersCount = 0; var onesCount = 0; var fivesCount = 0; var tensCount = 0; var twentiesCount = 0; var hundredsCount = 0; //calculate change and push to exactChange[] while (change - 100 >= 0 && hundreds >= 100) { change -= 100; hundreds -= 100; hundredsCount ++; } //add hundreds amount to exactChange[] if (hundredsCount > 0) { exactChange.push(["ONE HUNDRED",(100 * hundredsCount)]); } while (change - 20 >= 0 && twenties >= 20) { change -= 20; twenties -= 20; twentiesCount ++; } //add twenties amount to exactChange[] if (twentiesCount > 0) { exactChange.push(["TWENTY",(20 * twentiesCount)]); } while (change - 10 >= 0 && tens >= 10) { change -= 10; tens -= 10; tensCount ++; } //add tens amount to exactChange[] if (tensCount > 0) { exactChange.push(["TEN",(10 * tensCount)]); } while (change - 5 >= 0 && fives >= 5) { change -= 5; fives -= 5; fivesCount ++; } //add fives amount to exactChange[] if (fivesCount > 0) { exactChange.push(["FIVE",(5 * fivesCount)]); } while (change - 1 >= 0 && ones >= 1) { change -= 1; ones -= 1; onesCount ++; } //add ones amount to exactChange[] if (onesCount > 0) { exactChange.push(["ONE",(1 * onesCount)]); } while (change - 0.25 >= 0 && quarters >= 0.25) { change -= 0.25; quarters -= 0.25; quartersCount ++; } //add quarters amount to exactChange[] if (quartersCount > 0) { exactChange.push(["QUARTER",(0.25 * quartersCount)]); } while (change - 0.10 >= 0 && dimes >= 0.10) { change -= 0.10; dimes -= 0.10; dimesCount ++; } //add dimes amount to exactChange[] if (dimesCount > 0) { exactChange.push(["DIME",(0.1 * dimesCount)]); } while (change - 0.05 >= 0 && nickels >= 0.05) { change -= 0.05; nickels -= 0.05; nickelsCount ++; } //add nickels amount to exactChange[] if (nickelsCount > 0) { exactChange.push(["NICKEL",(0.05 * nickelsCount)]); } while (change - 0.01 >= 0 && pennies >= 0.01) { change -= 0.01; pennies -= 0.01; penniesCount ++; //make pennies and change an accurate number if (change > 0.000 && change < 0.01) { change = 0.01; } if (pennies > 0.000 && pennies < 0.01) { pennies = 0.01; } } //add pennies amount to exactChange[] if (penniesCount > 0) { exactChange.push(["PENNY",(0.01 * penniesCount)]); } //is exact change available? ecAvail = (change == 0); //not enough money for change or not exact change if (cidTotal < startChange | (!ecAvail)) { return {status: "INSUFFICIENT_FUNDS", change: []}; } //just enough to give change if (cidTotal == startChange) { return {status: "CLOSED", change: cid}; } //enough money for change and exact change if (cidTotal > startChange && (ecAvail)) { return {status: "OPEN", change: exactChange}; } }
