function palindrome(str) { var score = 0; // sort and filter str (find way to shorten code) var temp1 = str.toLowerCase().split(""); var forward = temp1.filter(function(temp1) {return /[\da-z]/.test(temp1);}); var temp2 = str.toLowerCase().split(""); var backward = temp2.filter(function(temp2) {return /[\da-z]/.test(temp2);}); backward.reverse(); // compare arrays for (var i = 0; i < forward.length; i++) { if (forward[i] === backward[i]) { score++; } } // are both arrays the same? return (score == forward.length); }
