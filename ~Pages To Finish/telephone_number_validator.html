function telephoneCheck(str) { //sort different character types into seperate arrays var digits = str.split(""); var number = digits.filter(function(digits) {return /[\d]/.test(digits);}); var bracketL = digits.filter(function(digits) {return /[\(]/.test(digits);}); var bracketR = digits.filter(function(digits) {return /[\)]/.test(digits);}); var dash = digits.filter(function(digits) {return /[-]/.test(digits);}); // ways a phone number should NOT be formatted (there's probably more) if (digits.length > 16 | digits.length < 10 | (number.length == 11 && digits[0] != 1) | number.length > 11 | number.length < 10 | bracketL.length > 1 | bracketR.length > 1 | bracketL.length != bracketR.length | (bracketL != 0 && number == 10 && digits[4] != bracketR[0]) | digits[digits.length-1] == bracketR[0] | dash.length > 2) { return false; } else { return true; } }
