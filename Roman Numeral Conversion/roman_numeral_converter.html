function convertToRoman(num) { var tempNum = num; var answer = []; var count = 0; function four() { answer.pop(); answer.pop(); answer.pop(); } while (tempNum >= 1000) { tempNum -= 1000; answer.push("M"); } while (tempNum >= 500) { tempNum -= 500; answer.push("D"); } while (tempNum >= 100) { tempNum -= 100; answer.push("C"); count++; if (count == 4) { four(); answer.push("D"); } } while (tempNum >= 50) { tempNum -= 50; answer.push("L"); } count = 0; while (tempNum >= 10) { tempNum -= 10; answer.push("X"); count++; if (count == 4) { four(); answer.push("L"); } } while (tempNum >= 5) { tempNum -= 5; answer.push("V"); } count = 0; while (tempNum >= 1) { tempNum -= 1; answer.push("I"); count++; if (count == 4) { four(); answer.push("V"); } } //fix VIV, LXL, DCD and return (find way to shorten code) return answer.join('').replace("VIV", "IX").replace("LXL", "XC").replace("DCD", "CM"); }
